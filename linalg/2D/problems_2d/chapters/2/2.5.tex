\renewcommand{\theequation}{\theenumi}
\begin{enumerate}[label=\arabic*.,ref=\thesubsection.\theenumi]
\item What lines are represented by the following equations:
\begin{multicols}{2}
\begin{enumerate}
{\footnotesize
\item 
$
xy=0
$
\item 
$
x^2-y^2 = 0
$
\item 
$
6x^2+xy-y^2 = 0
$
\item 
$
y^2 -2xy\tan\theta - x^2 = 0
$
\item 
$
x^3+3x^2y-3xy^2-y^3 = 0
$
}
\end{enumerate}
\end{multicols}
\item Find the angles between the pairs of straight lines represented by the following equations:
\begin{multicols}{2}
\begin{enumerate}
\item
$
x^2-4xy+y^2 = 0
$
\item
$
x^2-y^2 = 0
$
\item
$
x^2-5xy+4y^2 = 0
$
\item
$
x\brak{x+4y} = 0
$
\item
$
x^2+xy-6y^2 = 0
$
\end{enumerate}
\end{multicols}
\item Prove that the equations
\begin{align*}
x^2-4xy+y^2 = 0 \text{ and } x+y = 3
\end{align*}
are the sides of an equilateral triangle.
\end{enumerate}
